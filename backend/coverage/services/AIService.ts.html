import { GoogleGenerativeAI } from '@google/generative-ai';
import { AIServiceConfig } from '../types';
import { logger } from '../utils/logger';
 
export class AIService {
  private genAI: GoogleGenerativeAI;
  private model: string;
 
  constructor(config: AIServiceConfig) {
    this.genAI = new GoogleGenerativeAI(config.apiKey);
    this.model = config.model;
  }
 
  async processTranscript(transcript: string): Promise<string> {
    try {
      logger.debug('Processing transcript with AI service', {
        transcriptLength: transcript.length,
      });
 
      const model = this.genAI.getGenerativeModel({ model: this.model });
      const metaPrompt = this.buildPrompt(transcript);
 
      const result = await model.generateContent(metaPrompt);
      const response = await result.response;
      const text = response.text();
 
      logger.debug('AI service response received', { responseLength: text.length });
 
      return this.stripMarkdown(text);
    } catch (error) {
      logger.error('Failed to process transcript with AI service', error);
      throw new Error('AI service failed to process transcript');
    }
  }
 
  private buildPrompt(transcript: string): string {
    return `You are an expert productivity assistant. Your task is to process a raw, unstructured transcript of a voice note and organize it into a structured markdown document.
 
Analyze the following transcript:
---
${transcript}
---
 
Based on the text, perform the following actions and format your response in markdown:
 
1. **Categorize the content** by choosing the most appropriate category. You can use one of these common categories or create a new one if it better fits the content:
 
   Common categories:
   - projects: App ideas, software projects, creative projects, business ideas
   - daily: Todo items, shopping lists, reminders, daily tasks
   - personal: Health, fitness, habits, self-improvement, life goals
   - work: Work-related tasks, meetings, professional development
   - notes: General thoughts, observations, learning notes, random ideas
   
   Or create a new category that better describes the content (examples: travel, recipes, books, learning, finance, hobbies, etc.). Use single words or simple phrases, lowercase, no spaces (use hyphens if needed).
 
2. **Generate a concise title** for the entire thought process (5-10 words).
3. **Create a semantic filename** (lowercase, hyphens, no spaces, descriptive).
4. **Provide a 2-3 sentence summary** that captures the core essence of the ideas presented.
5. **Identify and list distinct "Ideas"**. For each idea, provide a short title and description.
6. **Extract all "Action Items"**. These are concrete tasks or things to do. If an action item is vague, rephrase it to be specific and actionable.
7. **Identify key "Keywords" or "Tags"** that categorize the content.
 
Format your response exactly like this markdown structure:
 
---
category: [category]
filename: [semantic-filename]
---
 
# [Title Here]
 
## Summary
[2-3 sentence summary here]
 
## Ideas
- **[Idea 1 Title]**: [Description]
- **[Idea 2 Title]**: [Description]
 
## Action Items
- [ ] [Specific actionable task 1]
- [ ] [Specific actionable task 2]
 
## Tags
[tag1], [tag2], [tag3]
 
Return ONLY the markdown content with no additional text or explanations.`;
  }
 
  private stripMarkdown(text: string): string {
    return text.replace(/```markdown\n|```\n|```/g, '').trim();
  }
}
